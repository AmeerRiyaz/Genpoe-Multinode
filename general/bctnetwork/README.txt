
Pre-requisite:
	- GO Lang
	- curl language
	- docker,docker-compose 




Script folder:
	network-config
		- Configuration file required BCTREST API to communicate with BCT Network
		
	docker-config:
		- contains the docker configuration file

	org-config:
		- contains configuration files of RevOrg, RegOrg, SurveyOrg & OtherOrg

	crypto-config
		- contains crypto configuration file for generating identities for all organisation

artifacts:
	- contains channel artificats
	- artifacts will be created, when we run the scripts

fabric_binaries
	-contains binaries of fabric
